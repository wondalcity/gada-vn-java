## `apps/mobile-shell/src/capacitor-init.ts`

Full initialization code — called once when the Capacitor WebView finishes loading the app shell:

```typescript
import { App } from '@capacitor/app'
import { Network } from '@capacitor/network'
import { PushNotifications } from '@capacitor/push-notifications'
import { StatusBar, Style } from '@capacitor/status-bar'
import { SplashScreen } from '@capacitor/splash-screen'
import { Keyboard } from '@capacitor/keyboard'
import { initSessionBridge } from './bridge/session-bridge'
import { initPushNotifications } from './native/push'

export async function initCapacitor() {
  // Set status bar to dark content (light background)
  await StatusBar.setStyle({ style: Style.Dark })
  await StatusBar.setBackgroundColor({ color: '#FFFFFF' })

  // Bridge Preferences → cookies for Next.js middleware
  await initSessionBridge()

  // Register push notifications
  await initPushNotifications()

  // Handle deep links
  App.addListener('appUrlOpen', ({ url }) => {
    handleDeepLink(url)
  })

  // Handle keyboard resize
  Keyboard.addListener('keyboardWillShow', ({ keyboardHeight }) => {
    document.body.style.paddingBottom = `${keyboardHeight}px`
  })
  Keyboard.addListener('keyboardWillHide', () => {
    document.body.style.paddingBottom = '0px'
  })

  // Network status banner
  Network.addListener('networkStatusChange', ({ connected }) => {
    const banner = document.getElementById('offline-banner')
    if (banner) banner.style.display = connected ? 'none' : 'flex'
  })

  // Hide splash screen after init
  await SplashScreen.hide()
}

function handleDeepLink(url: string) {
  try {
    const parsed = new URL(url)
    const locale = localStorage.getItem('gada_locale') ?? 'ko'
    if (parsed.protocol === 'gada:') {
      window.location.href = `/${locale}${parsed.pathname}${parsed.search}`
    }
  } catch {
    // Ignore invalid deep links
  }
}
```

## `apps/mobile-shell/src/bridge/session-bridge.ts`

```typescript
import { Preferences } from '@capacitor/preferences'

/**
 * Bridge: reads auth token from Capacitor Preferences and sets it as a cookie
 * so Next.js middleware can read it as 'gada_session'.
 *
 * Called once on app init before any navigation.
 */
export async function initSessionBridge(): Promise<void> {
  const { value: token } = await Preferences.get({ key: 'gada_session' })
  if (token) {
    // Set cookie that Next.js middleware reads
    document.cookie = `gada_session=${token}; path=/; SameSite=Strict`
  }

  const { value: locale } = await Preferences.get({ key: 'gada_locale' })
  if (locale && ['ko', 'vi', 'en'].includes(locale)) {
    localStorage.setItem('gada_locale', locale)
  }
}

/**
 * Called after Firebase login — stores token in both Preferences and cookie.
 */
export async function setSession(token: string): Promise<void> {
  await Preferences.set({ key: 'gada_session', value: token })
  document.cookie = `gada_session=${token}; path=/; SameSite=Strict`
}

/**
 * Called on logout — clears token from both Preferences and cookie.
 */
export async function clearSession(): Promise<void> {
  await Preferences.remove({ key: 'gada_session' })
  document.cookie = 'gada_session=; path=/; expires=Thu, 01 Jan 1970 00:00:00 GMT'
}
```

## `apps/mobile-shell/src/native/push.ts`

```typescript
import { PushNotifications } from '@capacitor/push-notifications'
import { Preferences } from '@capacitor/preferences'

export async function initPushNotifications(): Promise<void> {
  const permission = await PushNotifications.requestPermissions()
  if (permission.receive !== 'granted') return

  await PushNotifications.register()

  PushNotifications.addListener('registration', async ({ value: fcmToken }) => {
    await Preferences.set({ key: 'gada_fcm_token', value: fcmToken })

    // Register with backend (fire and forget — non-critical)
    const { value: sessionToken } = await Preferences.get({ key: 'gada_session' })
    if (sessionToken) {
      fetch('/api/v1/devices/fcm-token', {
        method: 'PUT',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${sessionToken}`,
        },
        body: JSON.stringify({ token: fcmToken }),
      }).catch(() => { /* retry on next launch */ })
    }
  })

  PushNotifications.addListener('pushNotificationActionPerformed', ({ notification }) => {
    const { type, data } = notification.data ?? {}
    navigateFromNotification(type, data)
  })
}

function navigateFromNotification(type: string, data: Record<string, string> = {}): void {
  const locale = localStorage.getItem('gada_locale') ?? 'ko'
  const routes: Record<string, string> = {
    application_status: `/${locale}/worker/applications`,
    contract_ready: `/${locale}/worker/contracts/${data.contractId ?? ''}`,
    attendance_recorded: `/${locale}/worker/hires`,
    manager_approved: `/${locale}/manager`,
    manager_rejected: `/${locale}/manager/profile`,
    hire_cancelled: `/${locale}/worker/hires`,
  }
  const path = routes[type] ?? `/${locale}/worker`
  window.location.href = path
}
```

## `apps/mobile-shell/src/native/camera.ts`

```typescript
import { Camera, CameraResultType, CameraSource } from '@capacitor/camera'

export interface CapturedImage {
  base64: string
  format: 'jpeg' | 'png'
  blob: Blob
  sizeKb: number
}

export async function captureIdDocument(side: 'front' | 'back'): Promise<CapturedImage> {
  const photo = await Camera.getPhoto({
    quality: 85,
    allowEditing: false,
    resultType: CameraResultType.Base64,
    source: CameraSource.Prompt,
  })

  const base64 = photo.base64String!
  const format = (photo.format as 'jpeg' | 'png') ?? 'jpeg'
  const blob = base64ToBlob(base64, `image/${format}`)

  if (blob.size > 10 * 1024 * 1024) {
    throw new Error('FILE_TOO_LARGE') // 10MB limit
  }

  return { base64, format, blob, sizeKb: Math.round(blob.size / 1024) }
}

export async function captureSiteImage(): Promise<CapturedImage> {
  const photo = await Camera.getPhoto({
    quality: 80,
    allowEditing: true,
    resultType: CameraResultType.Base64,
    source: CameraSource.Prompt,
  })

  const base64 = photo.base64String!
  const blob = base64ToBlob(base64, 'image/jpeg')

  if (blob.size > 10 * 1024 * 1024) {
    throw new Error('FILE_TOO_LARGE')
  }

  return { base64, format: 'jpeg', blob, sizeKb: Math.round(blob.size / 1024) }
}

function base64ToBlob(base64: string, mimeType: string): Blob {
  const byteString = atob(base64)
  const ab = new ArrayBuffer(byteString.length)
  const ia = new Uint8Array(ab)
  for (let i = 0; i < byteString.length; i++) {
    ia[i] = byteString.charCodeAt(i)
  }
  return new Blob([ab], { type: mimeType })
}
```

## Step-by-Step Setup Instructions

### Step 1 — Install dependencies

```bash
cd apps/mobile-shell
pnpm install
```

This installs all Capacitor plugins listed in `package.json`, `@capacitor/cli`, and `typescript`.

### Step 2 — Add Android platform

```bash
npx cap add android
```

This generates the `android/` directory with a full Gradle project. The directory is partially gitignored — `android/app/src/main/assets/public/` is populated at build time and should not be committed.

### Step 3 — Add iOS platform

```bash
npx cap add ios
```

This generates the `ios/` directory with an Xcode project. Requires macOS with Xcode 15+ installed. The `ios/App/App/public/` directory is populated at build time.

### Step 4 — Build Next.js static export

```bash
pnpm build:web
# Equivalent to: cd ../web-next && BUILD_TARGET=capacitor pnpm build
```

This triggers `output: 'export'` in `next.config.ts` (activated by the `BUILD_TARGET=capacitor` env var) and emits static HTML/JS/CSS to `apps/web-next/out/`. Only `(app)/` routes are exported — `(public)/` routes are served live from `https://gada.vn`.

### Step 5 — Sync to native

```bash
npx cap sync
```

This copies the `apps/web-next/out/` directory into both `android/app/src/main/assets/public/` and `ios/App/App/public/`, and updates native plugin configurations.

### Step 6 — Configure Firebase

Place the Firebase project configuration files in the correct native locations:

| File | Location | Source |
|---|---|---|
| `google-services.json` | `android/app/google-services.json` | Firebase Console → Project Settings → Android app |
| `GoogleService-Info.plist` | `ios/App/App/GoogleService-Info.plist` | Firebase Console → Project Settings → iOS app |

The Firebase app ID must match `vn.gada.app` (the `appId` in `capacitor.config.ts`). FCM is enabled automatically when these files are present.

### Step 7 — Configure Android release signing

Generate a keystore for release builds (do this once; store the keystore and passwords securely):

```bash
keytool -genkey -v \
  -keystore apps/mobile-shell/release.keystore \
  -alias gada-release \
  -keyalg RSA \
  -keysize 2048 \
  -validity 10000
```

Add the keystore passwords to `android/app/build.gradle` under the `signingConfigs` block, or pass them via environment variables in CI. The `keystorePath` and `keystoreAlias` in `capacitor.config.ts` reference this keystore.

The `release.keystore` file must never be committed to git — add it to `.gitignore`.

### Step 8 — Android permissions in `AndroidManifest.xml`

After `npx cap add android`, open `android/app/src/main/AndroidManifest.xml` and verify these `<uses-permission>` entries are present (Capacitor plugins add most of them automatically, but confirm):

```xml
<!-- Geolocation — show distance to job site -->
<uses-permission android:name="android.permission.ACCESS_COARSE_LOCATION" />
<uses-permission android:name="android.permission.ACCESS_FINE_LOCATION" />

<!-- Camera + Gallery — ID document upload, site images -->
<uses-permission android:name="android.permission.CAMERA" />
<uses-permission android:name="android.permission.READ_MEDIA_IMAGES" />
<!-- For Android < 13 (API 32 and below): -->
<uses-permission android:name="android.permission.READ_EXTERNAL_STORAGE"
    android:maxSdkVersion="32" />
<uses-permission android:name="android.permission.WRITE_EXTERNAL_STORAGE"
    android:maxSdkVersion="29" />

<!-- Push Notifications (FCM) — application status, contract ready, attendance -->
<uses-permission android:name="android.permission.POST_NOTIFICATIONS" />
<uses-permission android:name="android.permission.RECEIVE_BOOT_COMPLETED" />

<!-- Network — API calls, offline detection -->
<uses-permission android:name="android.permission.INTERNET" />
<uses-permission android:name="android.permission.ACCESS_NETWORK_STATE" />

<!-- Vibration — haptic feedback during signature drawing -->
<uses-permission android:name="android.permission.VIBRATE" />
```

Also add the deep link intent filter inside the `<activity>` element:

```xml
<intent-filter android:autoVerify="true">
  <action android:name="android.intent.action.VIEW" />
  <category android:name="android.intent.category.DEFAULT" />
  <category android:name="android.intent.category.BROWSABLE" />
  <data android:scheme="gada" android:host="*" />
  <data android:scheme="https" android:host="gada.vn" />
</intent-filter>
```

### Step 9 — iOS permissions in `Info.plist`

Open `ios/App/App/Info.plist` and add the following entries. Each description appears in the system permission dialog shown to the user — bilingual Korean/Vietnamese is required for the GADA VN audience:

```xml
<!-- Camera — ID document capture, site gallery -->
<key>NSCameraUsageDescription</key>
<string>신분증 및 현장 사진 업로드를 위해 카메라가 필요합니다. / Cần camera để tải lên ảnh CMND và công trường.</string>

<!-- Photo Library — pick existing images for ID or site gallery -->
<key>NSPhotoLibraryUsageDescription</key>
<string>신분증 및 현장 사진 업로드를 위해 사진 접근이 필요합니다. / Cần truy cập ảnh để tải lên ảnh CMND và công trường.</string>

<!-- Location — display distance to job site on job cards -->
<key>NSLocationWhenInUseUsageDescription</key>
<string>현장까지의 거리를 표시하기 위해 위치 정보가 필요합니다. / Cần vị trí để hiển thị khoảng cách đến công trường.</string>

<!-- Background push notifications (FCM via APNs) -->
<key>UIBackgroundModes</key>
<array>
  <string>remote-notification</string>
</array>
```

For Universal Links (HTTPS deep links), also add:

```xml
<key>com.apple.developer.associated-domains</key>
<array>
  <string>applinks:gada.vn</string>
</array>
```

And serve `https://gada.vn/.well-known/apple-app-site-association` from the Laravel backend.

### Step 10 — Development workflow

**Android (requires Android device or emulator connected via ADB, or emulator running):**

```bash
# From apps/mobile-shell
pnpm dev:android
# Equivalent to: npx cap run android --livereload --external
```

This starts the Next.js dev server, points the Capacitor WebView at it via `server.url` in `capacitor.config.ts`, and deploys to the connected device. Changes to `apps/web-next/` reload instantly via HMR.

Before running: update `server.url` in `capacitor.config.ts` to your machine's local IP address (`ifconfig | grep inet`).

**iOS (requires macOS + Xcode 15+ + iOS simulator or device):**

```bash
pnpm dev:ios
# Equivalent to: npx cap run ios --livereload --external
```

**Open native IDE for native-layer debugging:**

```bash
pnpm open:android   # Opens Android Studio
pnpm open:ios       # Opens Xcode
```

### Step 11 — Production release build

```bash
# Build and sync
pnpm build:web
npx cap sync

# Android — generates AAB (App Bundle) for Play Store
npx cap build android --prod
# Output: android/app/build/outputs/bundle/release/app-release.aab

# iOS — open Xcode and use Product → Archive
npx cap open ios
# Then in Xcode: Product → Archive → Distribute App → App Store Connect
```

The AAB format is required for Play Store uploads. Google Play applies device-specific optimization (splits APK by ABI/screen density), which reduces the end-user download size to under 12 MB.
